for..in Loops v1.5 NEW

The for keyword provides clean iteration over any array or string.

for item in collection { }

# Over array
for name in ["Alice", "Bob"] {
    print(name)
}

# Over range
for i in iter.range(10) {
    print(i)
}

# Over string characters
for ch in "hello" {
    print(ch)
}

const Variables v1.5 NEW

Constants are set once and never reassigned. Available everywhere in the current scope.

const MAX_SIZE = 1000
const APP_NAME = "MyApp"
const PI = 3.14159
const DEBUG = false

# const with type annotation
const LIMIT: int = 100

Structs & Impl

struct Point {
    x: float
    y: float
}

impl Point {
    do dist(self, other: Point) -> float {
        let dx = self.x - other.x
        let dy = self.y - other.y
        return math.sqrt(dx*dx + dy*dy)
    }
}

let p1 = Point { x: 0.0, y: 0.0 }
let p2 = Point { x: 3.0, y: 4.0 }
print(p1.dist(p2))  # 5.0

Decorators

# Memoize function results
@cache
do fib(n: int) -> int {
    if n <= 1 { return n }
    return fib(n-1) + fib(n-2)
}

# Time function execution
@timer
do slow_op() { }

# Mark as deprecated
@deprecated
do old_func() { }

VM & JIT

TSTNT v1.5 includes a bytecode compiler and stack VM. Functions are automatically JIT-compiled to ARM64 native code after 50 calls if they operate only on integers.

# This function will be JIT compiled on aarch64
do add(a: int, b: int) -> int {
    return a + b
}

# Benchmark output shows which backend is used
# $ tstnt bench fib.tstnt 10
# using VM + JIT

The VM uses indexed globals (Vec instead of HashMap) for O(1) variable access. Function calls avoid cloning AST thanks to Arc-backed chunks.
