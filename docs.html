v1.5.0 — What's New

Version 1.0 brings real decorator logic, new stdlib modules, 20+ new packages, and many new builtin functions.

@cache — real memoization

@cache
do fib(n: int) -> int {
  if n <= 1 { return n }
  return fib(n-1) + fib(n-2)
}
# fib(40) called once — subsequent calls use cache

@timer — real timing

@timer
do slow_work() {
  thread.sleep(200)
}
slow_work()
# @timer: slow_work() took 201ms

@deprecated — warning on call

@deprecated
do old_api() { }
old_api()
# warning: 'old_api' is deprecated

New stdlib modules

sql — flat-file relational tables  ·  server — HTTP server  ·  tui2 — full TUI framework  ·  result — Result/Option types  ·  fmt2 — formatting utilities

New builtin functions

find(arr, |x| x > 3)          # first match or null
every(arr, |x| x > 0)         # true if all match
any(arr, |x| x == 0)          # true if any match
flat_map(arr, |x| [x, x*2])   # map + flatten
count_where(arr, |x| x > 2)   # count matching
take(arr, 3)                    # first 3 elements
drop(arr, 2)                    # skip first 2
chunks(arr, 3)                  # split into chunks of 3
concat(a, b, c)                 # merge arrays
index_of(arr, val)              # -1 if not found
slice(arr, 1, 4)                 # arr[1..4]
sprintf("Hello {}", name)        # template fill

Decorators

Decorators modify function behaviour. TSTNT v1.0 has three built-in decorators with real logic.

# @cache — memoize by argument values
@cache
do expensive(n: int) -> int {
  thread.sleep(100)
  return n * n
}
expensive(5)  # computes, stores in cache
expensive(5)  # returns from cache instantly

# @timer — prints elapsed time after call
@timer
do render() {
  # ...
}
render()  # @timer: render() took 12ms

# @deprecated — prints warning on call
@deprecated
do legacy_parse(s: str) { }
legacy_parse("x")  # warning: 'legacy_parse' is deprecated

Structs & Impl

struct User {
  name: str
  age: int
}

impl User {
  do greet -> str {
    return "Hi, I am " + self.name
  }
  do is_adult -> bool {
    return self.age >= 18
  }
}

let u = User { name: "Alex", age: 25 }
print(u.greet())

Match

  100       -> print("perfect")
  80..99    -> print("great")
  60..79    -> print("ok")
  _         -> print("fail")
}

# Match with guard
match n {
  x if x < 0 -> print("negative")
  x if x > 0 -> print("positive")
  _ -> print("zero")
}

Lambdas & Functional

let double = |x| x * 2

let nums = [1,2,3,4,5]
let doubled  = map(nums, |x| x * 2)
let evens    = filter(nums, |x| x % 2 == 0)
let total    = reduce(nums, |acc x| acc + x, 0)
let first    = find(nums, |x| x > 3)
let ok       = every(nums, |x| x > 0)
let has_zero = any(nums, |x| x == 0)
let pairs    = flat_map(nums, |x| [x, x*2])

# Pipe operator
let result = value | double | str

SQL Module

Flat-file relational tables. No SQLite required. Works on Termux.

use sql

let conn = sql.open("myapp")
sql.create_table(conn, "users", ["id", "name", "age"])

sql.insert(conn, "users", ["1", "Alice", "25"])
sql.insert(conn, "users", ["2", "Bob", "30"])

let all = sql.select(conn, "users")
let alice = sql.where(conn, "users", "name", "Alice")

sql.update(conn, "users", "name", "Alice", "age", "26")
sql.delete_where(conn, "users", "name", "Bob")
print(sql.count(conn, "users"))

Result Module

use result

do divide(a: int, b: int) -> any {
  if b == 0 { return result.err("division by zero") }
  return result.ok(a / b)
}

let r = divide(10, 2)
if result.is_ok(r) {
  print(result.unwrap(r))
} else {
  print("Error: " + result.get_err(r))
}

let val = result.unwrap_or(divide(1, 0), 0)  # 0

# Option
let opt = result.some(42)
let empty = result.none()

Telegram Module

use tg
use thread

do main {
  tg.token("YOUR_BOT_TOKEN")
  tg.delete_webhook()

  let mut offset = 0
  while true {
    let updates = tg.get_updates(offset)
    loop upd in updates {
      offset = upd.update_id + 1
      if upd.text == "/start" {
        tg.send_keyboard(upd.chat_id, "Hello!",
          [["Option A", "Option B"]])
      }
    }
    thread.sleep(500)
  }
}
