TSTNT Documentation

Complete reference for the TSTNT programming language — v1.5.0.

v1.5.0 — What's New

Version 1.5.0 is the biggest release yet, adding JIT compilation, new syntax, three new stdlib modules, and 12 new packages.

New Syntax

# for..in loop
for item in [1, 2, 3] {
  print(item)
}

# const variables
const PI = 3.14159
const APP_NAME: str = "MyApp"

# Default params
do greet(name: str = "World") {
  print("Hello, {name}!")
}

New Stdlib Modules

# iter — iterator utilities
use "iter"
let pairs = iter.enumerate(["a", "b", "c"])
let r = iter.range(0, 10, 2)  # [0,2,4,6,8]
let w = iter.windows(arr, 3)

# bytes — binary data
use "bytes"
let b = bytes.from_str("hello")
let hex = bytes.to_hex(b)   # "68656c6c6f"

# reflect — runtime inspection
use "reflect"
print(reflect.type_of(42))   # "int"
print(reflect.fields(point)) # ["x","y"]

Control Flow

# if / else
if x > 0 {
  print("positive")
} else if x == 0 {
  print("zero")
} else {
  print("negative")
}

# while
let mut i = 0
while i < 10 { i += 1 }

# for..in (v1.5.0)
for item in items { print(item) }

# loop (range)
loop i in 0..10 { print(i) }

# loop in (foreach)
loop in items { print(item) }

# repeat
repeat 3 { print("hey") }

# ternary
let val = x > 0 ? "pos" : "neg"

Structs & Impl

struct Point {
  x: float
  y: float
}

impl Point {
  do dist(other: Point) -> float {
    let dx = self.x - other.x
    let dy = self.y - other.y
    return math.sqrt(dx*dx + dy*dy)
  }
  do str {
    return "({self.x}, {self.y})"
  }
}

let p = Point { x: 1.0, y: 2.0 }
print(p.str())

# unit (struct with methods inline)
unit Counter {
  value: int
  do inc { self.value += 1 }
}

Enums

  Shape::Circle(r) -> print("radius: {r}")
  Shape::Rect(w, h) -> print("{w}x{h}")
}

Pattern Matching

match value {
  1       -> print("one")
  2..5    -> print("two to five")
  x if x > 10 -> print("big: {x}")
  _       -> print("other")
}

# if let
if let val = maybe_value {
  print("got: {val}")
}

Lambdas & Pipes

let double = |x| x * 2
let nums = [1, 2, 3, 4, 5]
let evens = filter(nums, |x| x % 2 == 0)
let doubled = map(nums, |x| x * 2)
let total = reduce(nums, |acc, x| acc + x, 0)

# Pipe operator
let result = nums |> filter(|x| x > 2) |> map(|x| x * 10)

Decorators

@cache
do fib(n: int) -> int {
  if n <= 1 { return n }
  return fib(n-1) + fib(n-2)
}

@timer
do slow_op(n: int) { # prints elapsed ms
  loop i in 0..n { }
}

@deprecated
do old_func() { # prints warning on call
  print("old")
}

LSP

TSTNT includes a Language Server (tstnt-lsp) for editor integration.

# Neovim (init.lua)
vim.api.nvim_create_autocmd({ "BufRead", "BufNewFile" }, {
  pattern = "*.tstnt",
  callback = function()
    vim.lsp.start({
      name = "tstnt-lsp",
      cmd = { "tstnt-lsp" },
      root_dir = vim.fn.getcwd(),
    })
  end,
})

The LSP provides: autocomplete, hover documentation, go-to-definition, syntax errors.
