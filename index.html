Clean. Readable. Fast to write.

TSTNT v1.5 syntax is designed to be obvious at a glance.

# Variables & functions let name = "TSTNT" let mut count = 0 const MAX = 100 do add(a: int, b: int) -> int { return a + b } for i in [1, 2, 3] { print(i) } match count { 0 => print("zero") 1..10 => print("small") _ => print("big") }
struct User { name: str age: int } impl User { do greet(self) { print("Hi, I'm ${self.name}") } do is_adult(self) -> bool { return self.age >= 18 } } let u = User { name: "Alice", age: 30 } u.greet() print(u.is_adult())
# v1.5.0 new features const PI = 3.14159 for n in iter.range(0, 10) { print(n * n) } let pairs = iter.enumerate(["a", "b", "c"]) # [[0,"a"], [1,"b"], [2,"c"]] let b = bytes.from_str("hello") print(bytes.to_hex(b)) # 68656c6c6f print(reflect.type_of(42)) # int print(reflect.is_str("hi")) # true
use tg const TOKEN = env.get("BOT_TOKEN") do on_message(msg: str, chat_id: int) { if msg == "/start" { tg.send(TOKEN, chat_id, "Welcome! 🐉") } else { tg.send(TOKEN, chat_id, "Echo: ${msg}") } } tg.poll(TOKEN, on_message)

// GET STARTED

Install in minutes.

TSTNT runs anywhere Rust compiles. Including your Android phone via Termux.

01

Install Rust

pkg install rust

On Termux / Android. On Linux/Mac use rustup.rs

02

Clone & Build

git clone github.com/tstnt-lang/tstnt cargo build --release
03

Add to PATH

cp target/release/tstnt ~/bin/
04

Write code

tstnt hello.tstnt

Or start the REPL: tstnt repl
